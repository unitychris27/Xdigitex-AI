```python
def run_project():
    state = load_state()
    
    while state["pending"]:
        phase = state["pending"][0]
        prompt = PHASE_PROMPTS[phase].strip()
        
        print(f"\n{'='*50}")
        print(f"Running: {phase}")
        print(f"Prompt size: {len(prompt)} chars / ~{len(prompt)//4} tokens")
        print(f"{'='*50}")
        
        # Each phase: fresh context, tiny prompt
        try:
            result = safe_call_model(
                role=AgentRole.BUILDER,
                prompt=prompt
                # NO history, NO previous output
                # Clean slate every phase
            )
            
            # Execute the output via SSH
            ssh_result = execute_on_server(result["content"])
            
            # Verify before marking done
            if verify_phase(phase, ssh_result):
                state["completed"].append(phase)
                state["pending"].pop(0)
                save_state(state)
                print(f"✅ {phase} complete")
            else:
                # Move to failed, skip for now, continue
                state["failed"].append(phase)
                state["pending"].pop(0)
                save_state(state)
                print(f"⚠️ {phase} failed, continuing...")
                
        except Exception as e:
            print(f"❌ {phase} crashed: {e}")
            state["failed"].append(phase)
            state["pending"].pop(0)
            save_state(state)
            time.sleep(5)  # cooldown before next phase
            continue
    
    # After all phases, retry failed ones
    if state["failed"]:
        print(f"\nRetrying {len(state['failed'])} failed phases...")
        retry_failed(state)
    
    print("\n✅ TREEPLATFORM BUILD COMPLETE")
```
